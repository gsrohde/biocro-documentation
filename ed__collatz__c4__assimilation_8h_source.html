122 void ed_collatz_c4_assimilation::do_operation() const
123 {
124  // Calculate the temperature factor
125  const double temperature_factor = pow(*collatz_q10_ip, (*temperature_leaf_ip - 25.0) / 10.0);
126 
127  // Calculate the carbon-limited assimilation rate
128  const double co2_rate_constant = *collatz_k_ip * temperature_factor;
129  const double assimilation_carbon_limited = co2_rate_constant * *mole_fraction_co2_intercellular_ip;
130 
131  // Calculate the rubisco-limited assimilation rate
132  const double rubisco_capacity_base = *collatz_vmax_ip * temperature_factor;
133  const double rubisco_low_temp_inhibition = 1 + exp(0.3 * (*collatz_rubisco_temperature_lower_ip - *temperature_leaf_ip));
134  const double rubisco_high_temp_inhibition = 1 + exp(0.3 * (*temperature_leaf_ip - *collatz_rubisco_temperature_upper_ip));
135  const double assimilation_rubisco_limited = rubisco_capacity_base / (rubisco_low_temp_inhibition * rubisco_high_temp_inhibition);
136 
137  // Calculate the light-limited assimilation rate
138  const double assimilation_light_limited = *collatz_alpha_ip * *collatz_PAR_flux_ip;
139 
140  // Use the mixing parameter theta to determine the gross assimilation rate limited by rubisco and light
141  const double M_a = *collatz_theta_ip;
142  const double M_b = -1.0 * (assimilation_rubisco_limited + assimilation_light_limited);
143  const double M_c = assimilation_rubisco_limited * assimilation_light_limited;
144  const double M_root_term = M_b * M_b - 4 * M_a * M_c;
145 
146  double M;
147  if (M_a == 0) {
148  M = -M_c / M_b;
149  } else {
150  const double M_plus = (-M_b + sqrt(M_root_term)) / (2 * M_a);
151  const double M_minus = (-M_b - sqrt(M_root_term)) / (2 * M_a);
152  M = std::min(M_plus, M_minus);
153  }
154 
155  // Use the mixing parameter beta to determine the gross assimilation rate limited by carbon, rubisco, and light
156  const double A_a = *collatz_beta_ip;
157  const double A_b = -1.0 * (M + assimilation_carbon_limited);
158  const double A_c = M * assimilation_carbon_limited;
159  const double A_root_term = A_b * A_b - 4 * A_a * A_c;
160 
161  double A;
162  if (A_a == 0) {
163  A = -A_c / A_b;
164  } else {
165  const double A_plus = (-A_b + sqrt(A_root_term)) / (2 * A_a);
166  const double A_minus = (-A_b - sqrt(A_root_term)) / (2 * A_a);
167  A = std::min(A_plus, A_minus);
168  }
169 
170  // Calculate the temperature-dependent leaf respiration
171  const double respiration_base = *collatz_rd_ip * temperature_factor;
172  const double respiration_high_temp_inhibition = 1 + exp(0.3 * (*temperature_leaf_ip - 55.0));
173  const double respiration = respiration_base / respiration_high_temp_inhibition;
174 
175  // Calculate net assimilation rate
176  const double assimilation_net = A - respiration;
177 
178  // Apply a possible reduction accounting for the effect of water stress on the stomata
179  const double adjusted_assimilation_net = assimilation_net * *assimilation_adjustment_factor_WS_ip;
180 
181  // Check for error conditions
182  std::map<std::string, bool> errors_to_check = {
183  {"the quadratic M_a and M_b terms cannot both be zero", M_a == 0 && M_b == 0}, // divide by zero
184  {"the quadratic M_root_term cannot be negative for M_a != 0", M_a != 0 && M_root_term < 0}, // imaginary sqrt
185  {"the quadratic A_a and A_b terms cannot both be zero", A_a == 0 && A_b == 0}, // divide by zero
186  {"the quadratic A_root_term cannot be negative for A_a != 0", A_a != 0 && A_root_term < 0} // imaginary sqrt
187  };
188 
189  check_error_conditions(errors_to_check, get_name());
190 
191  // Update outputs
192  update(collatz_KT_op, co2_rate_constant);
193  update(collatz_RT_op, respiration);
194  update(collatz_VT_op, assimilation_rubisco_limited);
195  update(assimilation_carbon_limited_op, assimilation_carbon_limited);
196  update(assimilation_rubisco_limited_op, assimilation_rubisco_limited);
197  update(assimilation_light_limited_op, assimilation_light_limited);
198  update(assimilation_gross_op, A);
199  update(respiration_op, respiration);
200  update(assimilation_net_op, adjusted_assimilation_net);
201 }

159 void big_leaf_multilayer_canopy::do_operation() const {
160  // Collect inputs and make calculations
161  double lai = *lai_ip;
162  double n_layers = *nlayers_ip;
163  double rh = *rh_ip;
164  double windspeed = *windspeed_ip;
165  double kd = *kd_ip;
166  double lat = *lat_ip;
167  double doy_dbl = *doy_dbl_ip;
168  double solar = *solar_ip;
169  double chil = *chil_ip;
170 
171  const double kh = 1.0 - rh;
172  const double layer_lai = lai / n_layers;
173 
174  // Convert doy_dbl into doy and hour
175  int doy = floor(doy_dbl);
176  double hour = 24.0 * (doy_dbl - doy);
177 
178  // Define constants
179  constexpr double wind_speed_extinction = 0.7;
180  constexpr double leaf_radiation_absorptivity = 0.8;
181  constexpr double fraction_of_irradiance_in_PAR = 0.5; // dimensionless.
182  constexpr double joules_per_micromole_PAR = 0.235; // J / micromole. For the wavelengths that make up PAR in sunlight, one mole of photons has, on average, approximately 2.35 x 10^5 joules:
183 
184  // Get light properties
185  Light_model irradiance_fractions = lightME(lat, doy, hour);
186 
187  // TODO: The light at the top of the canopy should not be part of this function. They should be calculated before this is called.
188  // If cos(theta) is less than zero, the Sun is below the horizon, and lightME sets direct fraction to 0.
189  // Thus, calculation depending on direct_par_at_canopy_top should not need to check cos(theta), although other calcuations may need to.
190  const double direct_par_at_canopy_top = irradiance_fractions.direct_irradiance_fraction * solar;
191  const double diffuse_par_at_canopy_top = irradiance_fractions.diffuse_irradiance_fraction * solar;
192 
193  const double cosine_theta = irradiance_fractions.cosine_zenith_angle;
194  const double theta = acos(cosine_theta);
195  const double k0 = sqrt( pow(chil, 2) + pow(tan(theta), 2) );
196  const double k1 = chil + 1.744 * pow((chil + 1.183), -0.733);
197  const double light_extinction_k = k0 / k1; // Canopy extinction coefficient for an ellipsoidal leaf angle distribution. Page 251, Campbell and Norman. Environmental Biophysics. second edition.
198 
199  const double beam_radiation_at_canopy_top = direct_par_at_canopy_top * cosine_theta;
200  double mean_incident_direct_par = beam_radiation_at_canopy_top * light_extinction_k; // This is the same for all layers.
201 
202  // Initialize rates
203  double canopy_assimilation = 0;
204  double canopy_conductance = 0;
205  double canopy_transpiration = 0;
206  double sunlit_leaf_assimilation_rate;
207  double sunlit_leaf_stomatal_conductance;
208  double sunlit_leaf_transpiration_rate;
209  double shaded_leaf_assimilation_rate;
210  double shaded_leaf_stomatal_conductance;
211  double shaded_leaf_transpiration_rate;
212 
213  // Set some input parameters for the standalone modules that don't vary by layer
214  *theta_ptr = theta;
215 
216  // Iterate over the canopy layers
217  for (unsigned int n = 0; n < n_layers; ++n) {
218  // Get properties of this layer
219  const double CumLAI = layer_lai * (n + 1);
220  const double layer_wind_speed = windspeed * exp(-wind_speed_extinction * (CumLAI - layer_lai));
221 
222  const double j = n + 1; // Explicitly make j a double so that j / n_layers isn't truncated.
223  const double layer_relative_humidity = rh * exp(kh * (j / n_layers));
224 
225  const double layer_scattered_par = beam_radiation_at_canopy_top * (exp(-light_extinction_k * sqrt(leaf_radiation_absorptivity) * CumLAI) - exp(-light_extinction_k * CumLAI));
226  double layer_diffuse_par = diffuse_par_at_canopy_top * exp(-kd * CumLAI) + layer_scattered_par; // The exponential term is equation 15.6, pg 255 of Campbell and Normal. Environmental Biophysics. with alpha=1 and Kbe(phi) = Kd.
227 
228  double sunlit_lai = (1 - exp(-light_extinction_k * layer_lai)) * exp(-light_extinction_k * CumLAI) / light_extinction_k;
229 
230  if (cosine_theta <= 1e-10) {
231  // For values of cos(theta) close to or less than 0, the sun is below the horizon.
232  // In that case, no leaves are sunlit.
233  // I don't know how C++ handles that calculation since light_extinction_k would be infinite, so set sunlai_lai to 0 explicitly.
234  sunlit_lai = 0;
235  mean_incident_direct_par = direct_par_at_canopy_top / k1;
236  layer_diffuse_par = diffuse_par_at_canopy_top * exp(-light_extinction_k * CumLAI);
237  }
238 
239  const double shaded_lai = layer_lai - sunlit_lai;
240 
241  const double layer_shaded_par = layer_diffuse_par;
242  const double layer_sunlit_par = mean_incident_direct_par + layer_shaded_par;
243 
244  const double layer_sunlit_incident_irradiance = layer_sunlit_par * joules_per_micromole_PAR / fraction_of_irradiance_in_PAR; // W / m^2.
245  const double layer_shaded_incident_irradiance = layer_shaded_par * joules_per_micromole_PAR / fraction_of_irradiance_in_PAR; // W / m^2.
246 
247  // Set some input parameters for the standalone modules that do vary by layer
248  *layer_wind_speed_ptr = layer_wind_speed;
249  *layer_relative_humidity_ptr = layer_relative_humidity;
250 
251  // Run a calculation for the sunlit leaves and get the resulting rates
252  *Ipar_ptr = layer_sunlit_par;
253  *Irad_ptr = layer_sunlit_incident_irradiance;
254  canopy_modules->run();
255  sunlit_leaf_assimilation_rate = *leaf_ass_ptr;
256  sunlit_leaf_transpiration_rate = *leaf_trans_ptr;
257  sunlit_leaf_stomatal_conductance = *leaf_cond_ptr;
258 
259  // Run a calculation for the shaded leaves and get the resulting rates
260  *Ipar_ptr = layer_shaded_par;
261  *Irad_ptr = layer_shaded_incident_irradiance;
262  canopy_modules->run();
263  shaded_leaf_assimilation_rate = *leaf_ass_ptr;
264  shaded_leaf_transpiration_rate = *leaf_trans_ptr;
265  shaded_leaf_stomatal_conductance = *leaf_cond_ptr;
266 
267  // Add the results to the totals
268  canopy_assimilation += sunlit_lai * sunlit_leaf_assimilation_rate + shaded_lai * shaded_leaf_assimilation_rate; // micromoles / m^2 / s. Ground-area basis.
269  canopy_conductance += sunlit_lai * sunlit_leaf_stomatal_conductance + shaded_lai * shaded_leaf_stomatal_conductance; // mmol / m^2 / s. Ground-area basis.
270  canopy_transpiration += sunlit_lai * sunlit_leaf_transpiration_rate + shaded_lai * shaded_leaf_transpiration_rate; // kg / m^2 / s. Ground-area basis.
271  }
272 
273  // Update the output parameter list
274 
275  // Convert assimilation units
276  // 3600 - seconds per hour
277  // 1e-6 - moles per micromole
278  // 30 - grams per mole for CO2
279  // 1e-6 - megagrams per gram
280  // 10000 - meters squared per hectar
281 
282  // Convert transpiration units
283  // 3600 - seconds per hour
284  // 1e-3 - megagrams per kilogram
285  // 10000 - meters squared per hectare
286 
287  //TODO: Figure out why canopy_conductance is always 0. NOTE: I don't think this is true anymore (EBL)
288  update(canopy_assimilation_rate_op, canopy_assimilation * 3600 * 1e-6 * 30 * 1e-6 * 10000);
289  update(canopy_transpiration_rate_op, canopy_transpiration * 3600 * 1e-3 * 10000);
290  update(canopy_conductance_op, canopy_conductance);
291 }

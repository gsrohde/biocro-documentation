138 void c4_leaf_photosynthesis::do_operation() const
139 {
140  // Convert light inputs from energy to molecular flux densities
141  const double incident_par_micromol = incident_par / par_energy_content; // micromol / m^2 / s
142  const double incident_average_par_micromol = incident_average_par / par_energy_content; // micromol / m^2 / s
143 
144  // Get an initial estimate of stomatal conductance, assuming the leaf is at air temperature
145  const double initial_stomatal_conductance = c4photoC(
146  incident_par_micromol, temp, rh, vmax1, alpha1,
147  kparm, theta, beta, Rd, b0, b1, StomataWS, Catm,
148  water_stress_approach, upperT, lowerT)
149  .Gs; // mmol / m^2 / s
150 
151  // Calculate a new value for leaf temperature
152  double constexpr LAI = 0.0; // EvapoTrans2 does not actually use LAI for anything
153  const struct ET_Str et = EvapoTrans2(incident_par_micromol, incident_average_par_micromol, temp, rh, windspeed,
154  LAI, height, initial_stomatal_conductance, leafwidth, et_equation);
155 
156  const double leaf_temperature = temp + et.Deltat; // deg. C
157 
158  // Calculate final values for assimilation, stomatal conductance, and Ci using the new leaf temperature
159  const struct c4_str photo = c4photoC(
160  incident_par_micromol, leaf_temperature, rh, vmax1, alpha1,
161  kparm, theta, beta, Rd, b0, b1, StomataWS, Catm,
162  water_stress_approach, upperT, lowerT);
163 
164  // Update the outputs
165  update(Assim_op, photo.Assim);
166  update(GrossAssim_op, photo.GrossAssim);
167  update(Ci_op, photo.Ci);
168  update(Gs_op, photo.Gs);
169  update(TransR_op, et.TransR);
170  update(EPenman_op, et.EPenman);
171  update(EPriestly_op, et.EPriestly);
172  update(leaf_temperature_op, leaf_temperature);

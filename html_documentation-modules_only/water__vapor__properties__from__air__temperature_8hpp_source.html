58  // Collect input parameters and make calculations

60  double air_temperature = *temp_ip; // Degrees C

62 
63  double specific_heat_of_water = 1010; // J / kg / K
64  double molar_mass_of_water = 18.01528e-3; // kg / mol
65  double R = 8.314472; // joule / kelvin / mole
66 
67  double density_of_dry_air = TempToDdryA(air_temperature); // kg / m^3
68  double latent_heat_vaporization_of_water = TempToLHV(air_temperature); // J / kg
69  double saturation_water_vapor_pressure = saturation_vapor_pressure(air_temperature); // Pa
70  double saturation_water_vapor_content = saturation_water_vapor_pressure / R / (air_temperature + 273.15) * molar_mass_of_water; // kg / m^3. Convert from vapor pressure to vapor density using the ideal gas law. This is approximately right for temperatures what won't kill plants.
71  double vapor_density_deficit = saturation_water_vapor_content * (1 - rh);
72 
73  // Update the output parameter list
74  update(latent_heat_vaporization_of_water_op, latent_heat_vaporization_of_water); // J / kg
75  update(slope_water_vapor_op, TempToSFS(air_temperature)); // kg / m^3 / K. It is also kg / m^3 / degrees C since it's a change in temperature.
76  update(saturation_water_vapor_pressure_op, saturation_water_vapor_pressure); // Pa
77  update(water_vapor_pressure_op, saturation_water_vapor_pressure * rh); // Pa
78  update(vapor_density_deficit_op, vapor_density_deficit); // kg / m^3
79  update(psychrometric_parameter_op, density_of_dry_air * specific_heat_of_water / latent_heat_vaporization_of_water); // kg / m^3 / K

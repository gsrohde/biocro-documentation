136  // Pass the first guess to the observer
137  observer(initial_guess_for_root, false);
138 
139  // Initialize temporary variables for the loop
140  std::vector<double> previous_guess = initial_guess_for_root;
141  std::vector<double> next_guess = initial_guess_for_root;
142  bool need_to_make_adjustment;
143 
144  // Define a lambda to simplify error checking
145  auto errors_occurred = [](std::vector<bool> error_vector) -> bool {
146  int num_errors = std::accumulate(error_vector.begin(), error_vector.end(), 0);
147  return num_errors > 0;
148  };
149 
150  // Define a lambda to simplify convergence checking
151  auto check_convergence = [&errors_occurred](std::vector<bool> convergence_error_vector, bool* convergence_achieved) -> void {
152  if (!errors_occurred(convergence_error_vector)) {
153  *convergence_achieved = true;
154  }
155  };
156 
157  // Run the loop until convergence is achieved or the max number of iterations is exceeded
158  do {
159  ++num_iterations;
160 
161  next_guess = get_next_guess(se, previous_guess);
162 
163  need_to_make_adjustment = errors_occurred(is_outside_bounds(next_guess, lower_bounds, upper_bounds));
164 
165  if (need_to_make_adjustment) {
166  ++num_adjustments;
167  next_guess = adjust_bad_guess(next_guess, lower_bounds, upper_bounds);
168  } else {

170  {&converged_abs, has_not_converged_abs(previous_guess, next_guess, abs_error_tolerance)},
171  {&converged_rel, has_not_converged_rel(previous_guess, next_guess, rel_error_tolerance)}};
172 
173  for (auto const& x : convergence_checks) {
174  check_convergence(x.second, x.first);
175  }
176  }
177

78 void ed_evapotrans2::do_operation() const
79 {
80  // Collect inputs to EvapoTrans2 and convert units as necessary
81  const double Rad = 0.0; // micromoles / m^2 / s (only used for evapotranspiration)
82  const double airTemp = *temperature_air_ip; // degrees C
83  const double RH = *rh_ip; // dimensionless from Pa / Pa
84  const double WindSpeed = *windspeed_ip; // m / s
85  const double LeafAreaIndex = 0.0; // dimensionless from m^2 / m^2 (not actually used by EvapoTrans2)
86  const double CanopyHeight = 0.0; // meters (not actually used by EvapoTrans2)
87  const double stomatal_conductance = *conductance_stomatal_h2o_ip * 1e3; // mmol / m^2 / s
88  const double leaf_width = *leafwidth_ip; // meter
89  const int eteq = 0; // Report Penman-Monteith transpiration
90 
91  // For ed_penman_monteith_leaf_temperature, the light input is called `solar_energy_absorbed_leaf`
92  // and is in units of W / m^2 / s. This quantity is equivalent to `Ja2` in Evapotrans2. `Iave` is
93  // related to `Ja2` by:
94  // Ja2 = total_average_irradiance * (1 - LeafReflectance - tau) / (1 - tau)
95  // = Iave * joules_per_micromole_PAR / fraction_of_irradiance_in_PAR * (1 - LeafReflectance - tau) / (1 - tau)
96  // where:
97  // joules_per_micromole_PAR = 0.235
98  // fraction_of_irradiance_in_PAR = 0.5
99  // LeafReflectance = 0.2
100  // tau = 0.2
101  // So overall, Ja2 = Iave * 0.235 / 0.5 * (1 - 0.2 - 0.2) / (1 - 0.2)
102  // = Iave * 0.3525
103  // So we can find Iave according to Iave = Ja2 / 0.3525
104  const double Iave = *solar_energy_absorbed_leaf_ip / 0.3525; // micromoles / m^2 / s
105 
106  // Call EvapoTrans2
107  struct ET_Str et_results = EvapoTrans2(
108  Rad,
109  Iave,
110  airTemp,
111  RH,
112  WindSpeed,
113  LeafAreaIndex,
114  CanopyHeight,
115  stomatal_conductance,
116  leaf_width,
117  eteq);
118 
119  // Convert and return the results
120  update(evapotranspiration_penman_monteith_op, et_results.TransR * 1e-3); // mol / m^2 / s
121  update(evapotranspiration_penman_op, et_results.EPenman * 1e-3); // mol / m^2 / s
122  update(evapotranspiration_priestly_op, et_results.EPriestly * 1e-3); // mol / m^2 / s
123  update(temperature_leaf_op, airTemp + et_results.Deltat); // deg. C
124 }

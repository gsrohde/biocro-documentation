97 void calculate_jacobian(
98  equation_ptr_type const& equation_ptr,
99  x_vector_type const& x,
100  time_type t,
101  f_vector_type const& f_current,
102  matrix_type& jacobi)
103 {
104  size_t n = x.size();
105 
106  // Make a vector to store the perturbed f(x,t)
107  f_vector_type f_perturbed(n);
108 
109  // Perturb each element x_i of the input vector to find df_j(x,t)/dx_i,
110  // which is stored at jacobi(j,i)
111  double h;
112  x_vector_type x_perturbed = x;
113 
114  for (size_t i = 0; i < n; i++) {
115  // Ensure that the step size h is close to eps_deriv but is exactly representable
116  // (see Numerical Recipes in C, 2nd ed., Section 5.7)
117  h = calculation_constants::eps_deriv;
118  double temp = x[i] + h;
119  h = temp - x[i];
120 
121  // Calculate the new function value
122  x_perturbed[i] = x[i] + h; // Add h to the ith state variable
123  evaluate_equations(equation_ptr, x_perturbed, t, f_perturbed); // Calculate f_perturbed
124 
125  // Store the results in the Jacobian matrix
126  for (size_t j = 0; j < n; j++) {
127  jacobi(j, i) = (f_perturbed[j] - f_current[j]) / h;
128  }
129 
130  // Reset the ith state variable
131  x_perturbed[i] = x[i];
132  }
133 }

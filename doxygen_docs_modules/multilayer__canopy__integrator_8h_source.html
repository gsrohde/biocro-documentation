129 void multilayer_canopy_integrator::run() const
130 {
131  const double LAIc = lai / nlayers;
132  double canopy_assimilation_rate = 0;
133  double canopy_transpiration_rate = 0;
134  double canopy_conductance = 0;
135  double GrossAssim = 0;
136 
137  // Integrate assimilation, transpiration, and conductance throughout the canopy
138  for (int i = 0; i < nlayers; ++i) {
139  const double sunlit_lai = *sunlit_fraction_ips[i] * LAIc;
140  const double shaded_lai = *shaded_fraction_ips[i] * LAIc;
141 
142  canopy_assimilation_rate += *sunlit_Assim_ips[i] * sunlit_lai + *shaded_Assim_ips[i] * shaded_lai;
143  canopy_transpiration_rate += *sunlit_TransR_ips[i] * sunlit_lai + *shaded_TransR_ips[i] * shaded_lai;
144  canopy_conductance += *sunlit_Gs_ips[i] * sunlit_lai + *shaded_Gs_ips[i] * shaded_lai;
145  GrossAssim += *sunlit_GrossAssim_ips[i] * sunlit_lai + *shaded_GrossAssim_ips[i] * shaded_lai;
146  }
147 
148  // Modify net assimilation to account for respiration
149  // Note: this was originally only done for the C3 canopy
150  // Note: it seems like this should not be necessary since the assimilation model includes
151  // respiration
152  canopy_assimilation_rate *= (1.0 - growth_respiration_fraction);
153 
154  // Convert assimilation values from micromol / m^2 / s to Mg / ha / hr using the following
155  // conversion factors:
156  // 3600 - seconds per hour
157  // 1e-6 - moles per micromole
158  // 30 - grams of C6H12O6 (glucose) incorporated into dry biomass per mole of CO2
159  // 1e-6 - megagrams per gram
160  // 10000 - meters squared per hectare
161 
162  // Convert transpiration values from mol / m^2 / s to Mg / ha / hr using the following
163  // conversion factors:
164  // 3600 - seconds per hour
165  // 1e-3 - millimoles per mole
166  // 18 - grams per mole for H2O
167  // 1e-6 - megagrams per gram
168  // 10000 - meters squared per hectare
169 
170  update(canopy_assimilation_rate_op, canopy_assimilation_rate * 3600 * 1e-6 * 30 * 1e-6 * 10000);
171  update(GrossAssim_op, GrossAssim * 3600 * 1e-6 * 30 * 1e-6 * 10000);
172  update(canopy_transpiration_rate_op, canopy_transpiration_rate * 3600 * 1e-3 * 18 * 1e-6 * 10000);
173  update(canopy_conductance_op, canopy_conductance);
174 }

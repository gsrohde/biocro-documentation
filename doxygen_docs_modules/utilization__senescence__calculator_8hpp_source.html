73  double kStem = *rate_constant_stem_senescence_ip;

77  double Leaf = *Leaf_ip;
78  double Stem = *Stem_ip;
79  double Root = *Root_ip;
80  double Rhizome = *Rhizome_ip;
81 
82  // In the Thornley senescence model, sensescence is proportional to structural mass
83  // e.g. senescence_leaf = kLeaf * Leaf, for 0 < kLeaf < 1
84  // This is quite general. To see why, imagine using Michaelis-Menton kinetics here.
85  // In this case, we would have a catalyzed reaction turning structural carbon into
86  // substrate or litter
87  // The rate of change of the structural carbon concentration would be
88  // proportional to [structural carbon]/(Km + [structural carbon])
89  // And so the rate of change of total structural carbon would be proportional
90  // to volume * [structural carbon]/(Km + [structural carbon])
91  // For a tissue, volume is proportional to the amount of structural carbon
92  // In this case, [structural carbon] is constant
93  // E.g. we have a rate proportional to Leaf * constant/(Km + constant)
94  // We can absorb the constant into the proportionality constant, ending up with
95  // senescence_leaf = kLeaf * Leaf again
96  // So ultimately this is the same as the simpler assumption
97  // Here we do not allow the grain to senesce
98 
99  double senescence_leaf = kLeaf * Leaf;
100  double senescence_stem = kStem * Stem;
101  double senescence_root = kRoot * Root;
102  double senescence_rhizome = kRhizome * Rhizome;
103 
104  // Update the output parameter list
105  update(senescence_leaf_op, senescence_leaf);
106  update(senescence_stem_op, senescence_stem);
107  update(senescence_root_op, senescence_root);
108  update(senescence_rhizome_op, senescence_rhizome);
109 }

136 void collatz_leaf::do_operation() const {
137  // Collect inputs and make calculations
138  double incident_irradiance = *incident_irradiance_ip;
139  double incident_par = *incident_par_ip;
140  double temp = *temp_ip; // air temperature
141  double vmax = *vmax_ip;
142  double alpha = *alpha_ip;
143  double kparm = *kparm_ip;
144  double theta = *theta_ip;
145  double beta = *beta_ip;
146  double Rd = *Rd_ip;
147  double upperT = *upperT_ip;
148  double lowerT = *lowerT_ip;
149  double k_Q10 = *k_Q10_ip;
150  double Catm = *Catm_ip;
151  double rh = *rh_ip;
152  double b0 = *b0_ip;
153  double b1 = *b1_ip;
154  double layer_wind_speed = *layer_wind_speed_ip;
155  double leafwidth = *leafwidth_ip;
156  double water_vapor_pressure = *water_vapor_pressure_ip;
157  double stefan_boltzman = *stefan_boltzman_ip;
158  double leaf_reflectance = *leaf_reflectance_ip;
159  double leaf_transmittance = *leaf_transmittance_ip;
160  double water_stress_approach = *water_stress_approach_ip;
161  double StomataWS = *StomataWS_ip;
162 
163  double constexpr volume_of_one_mole_of_air = 24.39e-3; // m^3 / mol. TODO: This is for about 20 degrees C at 100000 Pa. Change it to use the model state. (1 * R * temperature) / pressure
164 
165  // Initial guesses
166  double intercelluar_co2_molar_fraction = Catm * 0.4; // micromole / mol
167 
168  double leaf_temperature = temp;
169 
170  struct collatz_result r = collatz_photo(incident_par, leaf_temperature, vmax,
171  alpha, kparm, theta, beta, Rd,
172  upperT, lowerT, k_Q10, intercelluar_co2_molar_fraction);
173 
174  double gs = ball_berry(r.assimilation * 1e-6, Catm * 1e-6, rh, b0, b1); // mmol / m^2 / s
175  double conductance_in_m_per_s = gs * 1e-3 * volume_of_one_mole_of_air; // m / s
176 
177  double boundary_layer_conductance = leaf_boundary_layer_conductance(layer_wind_speed,
178  leafwidth, temp, leaf_temperature - temp, conductance_in_m_per_s, water_vapor_pressure); // m / s
179 
180  double absorbed_irradiance = incident_irradiance * (1 - leaf_reflectance - leaf_transmittance) / (1 - leaf_transmittance); // W / m^2. Leaf area basis.
181 
182  double black_body_radiation = 0; // W / m^2. Leaf area basis.
183 
184  double leaf_net_irradiance = absorbed_irradiance; // W / m^2. Leaf area basis.
185 
186  // Convergence iteration
187  unsigned int constexpr max_iterations = 50;
188  for (unsigned int n = 0; n < max_iterations; ++n) {
189  // Store the previous assimilation rate
190  double const previous_assimilation = r.assimilation; // micromole / m^2 / s
191 
192  // Collect new inputs for the penman monteith leaf temperature model
193  leaf_net_irradiance = absorbed_irradiance - black_body_radiation; // W / m^2. Leaf area basis.
194 
195  // Update the penman monteith leaf temperature module input parameter values
196  *gs_ptr = gs;
197  *boundary_layer_conductance_ptr = boundary_layer_conductance;
198  *leaf_net_irradiance_ptr = leaf_net_irradiance;
199 
200  // Get the new leaf temperature
201  leaf_temperature_module->run();
202  leaf_temperature = *leaf_temperature_ptr;
203 
204  // Update Ci
205  double constexpr ratio = 1.6; // The ratio of the diffusivities of H2O and CO2 in air.
206  intercelluar_co2_molar_fraction = Catm - r.assimilation / gs * 1e3 * ratio; // micromole / mol
207  intercelluar_co2_molar_fraction = fmax(0, intercelluar_co2_molar_fraction);
208 
209  // Calculate new assimilation rate
210  r = collatz_photo(incident_par, leaf_temperature, vmax,
211  alpha, kparm, theta, beta, Rd,
212  upperT, lowerT, k_Q10, intercelluar_co2_molar_fraction);
213 
214  // Apply water stress
215  if (water_stress_approach == 0) r.assimilation *= StomataWS;
216  if (water_stress_approach == 1) gs *= StomataWS;
217 
218  // Check for stability
219  if (n > max_iterations - 10)
220  gs = b0 * 1e3; // mmol / m^2 / s. If it has gone through this many iterations, the convergence is not stable. This convergence is inapproriate for high water stress conditions, so use the minimum gs to try to get a stable system.
221 
222  // Update other things
223  conductance_in_m_per_s = gs * 1e-3 * volume_of_one_mole_of_air; // m / s
224 
225  boundary_layer_conductance = leaf_boundary_layer_conductance(layer_wind_speed, leafwidth, temp, leaf_temperature - temp, conductance_in_m_per_s, water_vapor_pressure); // m / s
226 
227  double delta_t = leaf_temperature - temp;
228 
229  black_body_radiation = 4 * stefan_boltzman * pow(273.15 + temp, 3) * delta_t; // W / m^2. Leaf area basis.
230 
231  // black_body_radiation = net long wave radiation emittted per second = radiation emitted per second - radiation absorbed per second = sigma * (Tair + deltaT)^4 - sigma * Tair^4
232  // To make it a linear function of deltaT, do a Taylor series about deltaT = 0 and keep only the zero and first order terms.
233  // black_body_radiation = sigma * (Tair - 0)^4 - sigma * Tair^4 + deltaT * (4 * sigma * Tair^3) = 4 * sigma * Tair^3 * deltaT
234  // where 4 * sigma * Tair^3 is the derivative of sigma * (Tair + deltaT)^4 evaluated at deltaT = 0
235 
236  double constexpr tol = 0.01; // micromole / m^2 / s
237  double const diff = fabs(previous_assimilation - r.assimilation);
238  if (diff <= tol) break;
239  }
240 
241  // Update the output parameter list
242  update(leaf_assimilation_rate_op, r.assimilation);
243  update(leaf_stomatal_conductance_op, gs);
244  update(leaf_boundary_layer_conductance_op, boundary_layer_conductance);
245  update(leaf_temperature_op, leaf_temperature);
246  update(Ci_op, intercelluar_co2_molar_fraction);
247  update(leaf_net_irradiance_op, leaf_net_irradiance);
248 }

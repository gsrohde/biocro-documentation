141 void two_layer_soil_profile::do_operation() const {
142  // Collect inputs and make calculations
143  double cws1 = *cws1_ip;
144  double cws2 = *cws2_ip;
145  double soil_depth1 = *soil_depth1_ip;
146  double soil_depth2 = *soil_depth2_ip;
147  double soil_depth3 = *soil_depth3_ip;
148  double soil_water_content = *soil_water_content_ip;
149 
150  double cws[] = {cws1, cws2};
151  double soil_depths[] = {soil_depth1, soil_depth2, soil_depth3};
152 
153  struct soilML_str soilMLS = soilML(*precip_ip, *canopy_transpiration_rate_ip, cws, soil_depth3, soil_depths,
154  *soil_field_capacity_ip, *soil_wilting_point_ip, *soil_saturation_capacity_ip, *soil_air_entry_ip, *soil_saturated_conductivity_ip,
155  *soil_b_coefficient_ip, *soil_sand_content_ip, *phi1_ip, *phi2_ip, *wsFun_ip,
156  2 /* Always uses 2 layers. */, *Root_ip, *lai_ip, 0.68, *temp_ip,
157  *solar_ip, *windspeed_ip, *rh_ip, *hydrDist_ip, *rfl_ip,
158  *rsec_ip, *rsdf_ip, *soil_clod_size_ip, *soil_reflectance_ip, *soil_transmission_ip,
159  *specific_heat_ip, *stefan_boltzman_ip);
160 
161  double layer_one_depth = soil_depth2 - soil_depth1;
162  double layer_two_depth = soil_depth3 - soil_depth2;
163  double cws_mean = (soilMLS.cws[0] * layer_one_depth + soilMLS.cws[1] * layer_two_depth) / (layer_one_depth + layer_two_depth);
164 
165  // Update the output parameter list
166  update(cws1_op, soilMLS.cws[0] - cws1);
167  update(cws2_op, soilMLS.cws[1] - cws2);
168  update(soil_water_content_op, cws_mean - soil_water_content);
169 }

161  double dLeafLitter = 0.0, dStemLitter = 0.0, dRootLitter = 0.0, dRhizomeLitter = 0.0;
162  double dstem_senescence_index = 0.0, droot_senescence_index = 0.0, drhizome_senescence_index = 0.0;
163 
164  // Calculate the leaf death rate due to frost
165  double frost_leaf_death_rate = 0.0;
166  if(TTc >= seneLeaf) { // Leaf senescence has started
167  bool A = lat >= 0.0; // In Northern hemisphere
168  bool B = doy_dbl >= 180.0; // In second half of the year
169  if((A && B) || ((!A) && (!B))) { // Winter in either hemisphere
170  // frost_leaf_death_rate changes linearly from 100 to 0 as temp changes from Tfrostlow to Tfrosthigh and is limited to [0,100]
171  std::max(0.0, std::min(100.0, 100.0 * (Tfrosthigh - temp) / (Tfrosthigh - Tfrostlow)));
172  }
173  }
174 
175  // The current leaf death rate is the larger of the previously stored leaf death rate and the new frost death rate
176  // I.e., the leaf death rate sometimes increases but never decreases
177  double current_leaf_death_rate = std::max(leafdeathrate, frost_leaf_death_rate);
178 
179  // Report the change in leaf death rate as the derivative of leafdeathrate
180  // Note: this will probably only work well with the Euler method
181  dLeafdeathrate = current_leaf_death_rate - leafdeathrate;
182 
183  // Calculate leaf senescence
184  if(TTc >= seneLeaf) {
185  // Use the leaf death rate to determine the change in leaf mass
186  double change = Leaf * current_leaf_death_rate * (0.01 / 24);
187 
188  // Remobilize some of the lost leaf tissue and send the rest to the litter
189  double remob = 0.6;
190  dLeaf += -change + kLeaf * change * remob; // Why does the leaf remobilize its own tissue? (EBL)
191  dLeafLitter += change * (1.0 - remob);
192  dRhizome += kRhizome * change * remob;
193  dStem += kStem * change * remob;
194  dRoot += kRoot * change * remob;
195  dGrain += kGrain * change * remob;
196  }
197 
198  // Calculate stem senescence
199  if(TTc >= seneStem) {
200  // Look back in time to find out how much the tissue grew in the past.
201  double change = newStem_vec[stem_senescence_index];
202 
203  // Subtract the new growth from the tissue derivative
204  dStem -= change;
205 
206  // Send the lost tissue to the litter
207  dStemLitter += change;
208 
209  // Increment the tissue senescence index
210  dstem_senescence_index++;
211  }
212 
213  // Calculate root senescence
214  if(TTc >= seneRoot) {
215  // Look back in time to find out how much the tissue grew in the past.
216  double change = newRoot_vec[root_senescence_index];
217 
218  // Subtract the new growth from the tissue derivative
219  dRoot -= change;
220 
221  // Send the lost tissue to the litter
222  dRootLitter += change;
223 
224  // Increment the tissue senescence index
225  droot_senescence_index++;
226  }
227 
228  // Calculate rhizome senescence
229  if(TTc >= seneRhizome) {
230  // Look back in time to find out how much the tissue grew in the past.
231  double change = newRhizome_vec[rhizome_senescence_index];
232 
233  // Subtract the new growth from the tissue derivative
234  dRhizome -= change;
235 
236  // Send the lost tissue to the litter
237  dRhizomeLitter += change;
238 
239  // Increment the tissue senescence index
240  drhizome_senescence_index++;
241  }
242

244

102 void utilization_growth::do_operation() const {
103  // Collect inputs and make calculations
104  double canopy_assimilation_rate = *canopy_assimilation_rate_ip;
105 
106  double utilization_leaf = *utilization_leaf_ip;
107  double utilization_stem = *utilization_stem_ip;
108  double utilization_grain = *utilization_grain_ip;
109  double utilization_root = *utilization_root_ip;
110  double utilization_rhizome = *utilization_rhizome_ip;
111 
112  double transport_leaf_to_stem = *transport_leaf_to_stem_ip;
113  double transport_stem_to_grain = *transport_stem_to_grain_ip;
114  double transport_stem_to_root = *transport_stem_to_root_ip;
115  double transport_stem_to_rhizome = *transport_stem_to_rhizome_ip;
116 
117  // The rate of change of the carbon substrate pool in each tissue is the sum of its inputs and outputs
118  // The leaf gets carbon from canopy photosynthesis (a conceptual rather than physical distinction), uses some for growth, and transfers some to the stem
119  // The stem gets carbon from the leaf, uses some for growth, and transfers some to the root, rhizome, and grain
120  // The root, rhizome, and grain each get carbon from the stem and use some for growth
121  // See Thornley, J. H. M. A Model to Describe the Partitioning of Photosynthate during Vegetative Plant Growth. Ann Bot 36, 419430 (1972)
122 
123  double d_substrate_leaf = canopy_assimilation_rate - transport_leaf_to_stem - utilization_leaf;
124  double d_substrate_stem = transport_leaf_to_stem - transport_stem_to_grain - transport_stem_to_root - transport_stem_to_rhizome - utilization_stem;
125  double d_substrate_grain = transport_stem_to_grain - utilization_grain;
126  double d_substrate_root = transport_stem_to_root - utilization_root;
127  double d_substrate_rhizome = transport_stem_to_rhizome - utilization_rhizome;
128 
129  // In Thornley's model, the utilized carbon substrate is split into two parts:
130  // (1) A fraction YG is used for growth (0 < YG < 1)
131  // (2) The remaining fraction (1 - YG) is lost to respiration
132  // In BioCro, the canopy assimilation rate (used here as the carbon input) already includes a negative
133  // respiratory contribution based on Bernacchi et al., Plant, Cell & Environment 24, 253259 (2001)
134  // So we simply set YG = 1 when allocating the substrate for growth, i.e.,
135  // the rate of change of Leaf is utilization_leaf, etc
136 
137  // Update the output parameter list
138  update(Leaf_op, utilization_leaf);
139  update(substrate_pool_leaf_op, d_substrate_leaf);
140 
141  update(Stem_op, utilization_stem);
142  update(substrate_pool_stem_op, d_substrate_stem);
143 
144  update(Grain_op, utilization_grain);
145  update(substrate_pool_grain_op, d_substrate_grain);
146 
147  update(Root_op, utilization_root);
148  update(substrate_pool_root_op, d_substrate_root);
149 
150  update(Rhizome_op, utilization_rhizome);
151  update(substrate_pool_rhizome_op, d_substrate_rhizome);
152 }

217  double substrate_pool_grain = *substrate_pool_grain_ip;
218 
219  // Create a logistic scaling coefficient to smoothly ramp up the grain growth rate
220  // The midpoint is a TTc value defined in the input parameters
221  // The maximum value is 1
222  // The steepness is chosen so the coefficient reaches 95% of its maximum
223  // value when TTc passes the midpoint by 8 thermal days:
224  // k = ln(0.95/(1-0.95))/8 = 0.368
225  // This roughly corresponds to grain growth "turning on" over the course of
226  // 2*8 = 16 thermal days
227 
228  double kGrain_scale = 1/(1 + exp(-0.368*(TTc - grain_TTc)));
229 
230  // Create an exponential scaling coefficient to smoothly ramp down the rhizome
231  // "retranslocation" rate
232 
233  //double kRhizome_retrans = 0.01 * kRhizome * exp(-TTc/300.0);
234  double kRhizome_retrans = 0.0;
235 
236  // Calculate the substrate utilization rates
237  // mass_fraction_leaf = substrate_pool_leaf / Leaf is a measure of the substrate concentration in the leaf,
238  // made under the assumption that leaf volume is proportional to its structural carbon component, i.e., Leaf
239  // We assume there is a catalyzed reaction transforming the substrate carbon into structural carbon.
240  // Using Michaelis-Menton kinetics, the rate of change of the structural carbon concentration is proportional
241  // to [substrate]/(Km + [substrate]), i.e., mass_fraction_leaf / (KmLeaf + mass_fraction_leaf)
242  // The total substrate utilization is this rate multiplied by the leaf volume, i.e.
243  // kLeaf * Leaf * mass_fraction_leaf / (KmLeaf + mass_fraction_leaf) = kLeaf * substrate_pool_leaf / (KmLeaf + mass_fraction_leaf)
244  // where kLeaf includes the various proportionality constants mentioned previously
245  // See Thornley, J. H. M. A Model to Describe the Partitioning of Photosynthate during Vegetative Plant Growth. Ann Bot 36, 419430 (1972)
246  // Alternatively, we could use a Hill equation to model utilization. This may be especially helpful for the leaf, since utilization
247  // can be made to fall off faster at small substrate concentrations. Note that the M-M equation is a special case of the Hill
248  // equation when n = 1.
249 
250  double n = 2.0; // Set to 1.0 for Michaelis-Morley kinetics. Use larger values to get a steeper utilization cutoff at small substrate concentrations.
251 
252  double mass_fraction_leaf = substrate_pool_leaf / Leaf;
253  double utilization_leaf = kLeaf * Leaf * utilization_hill(KmLeaf, n, mass_fraction_leaf);
254 
255  double mass_fraction_stem = substrate_pool_stem / Stem;
256  double utilization_stem = kStem * Stem * utilization_hill(KmStem, n, mass_fraction_stem);
257 
258  double mass_fraction_root = substrate_pool_root / Root;
259  double utilization_root = kRoot * Root * utilization_hill(KmRoot, n, mass_fraction_root);
260 
261  double mass_fraction_rhizome = substrate_pool_rhizome / Rhizome;
262  double utilization_rhizome = kRhizome * Rhizome * utilization_hill(KmRhizome, n, mass_fraction_rhizome);
263  utilization_rhizome -= kRhizome_retrans * Rhizome; // Add an additional term that allows the rhizome to redistribute some of its mass in the early stages of growth, i.e., when TTc is small
264 
265  double mass_fraction_grain = substrate_pool_grain / Grain;
266  double utilization_grain = kGrain * kGrain_scale * Grain * utilization_hill(KmGrain, n, mass_fraction_grain);
267 
268  // Calculate the transport rates
269  // We make the simple assumption that substrate transport is proportional to a concentration gradient between tissues
270  // and inversely proportional to a resistance, analogous to current in an electrical circuit
271  // It is important to scale the transport as the plant grows. For example, if the plant uniformly doubles in volume,
272  // the roots now require twice as much carbon substrate to continue growth at the same rate, so the resistance between
273  // the stem and root should be halved (or transport should double). We accomplish this scaling with the beta factor.
274  // See Thornley, J. H. M. A Model to Describe the Partitioning of Photosynthate during Vegetative Plant Growth. Ann Bot 36, 419430 (1972)
275 
276  double beta = Leaf + Grain + Stem + Root + Rhizome;
277 
278  double transport_leaf_to_stem = beta * (mass_fraction_leaf - mass_fraction_stem) / resistance_leaf_to_stem;
279  double transport_stem_to_grain = beta * (mass_fraction_stem - mass_fraction_grain) / resistance_stem_to_grain;
280  double transport_stem_to_root = beta * (mass_fraction_stem - mass_fraction_root) / resistance_stem_to_root;
281  double transport_stem_to_rhizome = beta * (mass_fraction_stem - mass_fraction_rhizome) / resistance_stem_to_rhizome;
282 
283  // Update the output parameter list
284  update(utilization_leaf_op, utilization_leaf);
285  update(utilization_stem_op, utilization_stem);
286  update(utilization_root_op, utilization_root);
287  update(utilization_rhizome_op, utilization_rhizome);
288  update(utilization_grain_op, utilization_grain);

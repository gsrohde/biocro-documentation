69 void bucket_soil_drainage::do_operation() const {
70  // Collect inputs and make calculations
71  double soil_field_capacity = *soil_field_capacity_ip; // m^3 / m^3.
72  double soil_wilting_point = *soil_wilting_point_ip; // m^3 / m^3.
73  double soil_water_content = *soil_water_content_ip; // m^3 / m^3.
74  double soil_saturation_capacity = *soil_saturation_capacity_ip; // m^3 / m^3.
75  double soil_depth = *soil_depth_ip; // meters
76  double precipitation_rate = *precipitation_rate_ip; // m / s.
77  double soil_saturated_conductivity = *soil_saturated_conductivity_ip;
78  double soil_air_entry = *soil_air_entry_ip;
79  double soil_b_coefficient = *soil_b_coefficient_ip;
80  double canopy_transpiration_rate = *canopy_transpiration_rate_ip;
81  double soil_evaporation_rate = *soil_evaporation_rate_ip;
82 
83  constexpr double g = 9.8; // m / s^2. Acceleration due to gravity.
84  constexpr double density_of_water_at_20_celcius = 998.2; // kg m^-3.
85 
86  /* soil_matric_potential is calculated as per "Dynamic Simulation of Water Deficit Effects upon Maize Yield" R. F. Grant Agricultural Systems. 33(1990) 13-39. */
87  double soil_matric_potential = -exp(log(0.033) + log(soil_field_capacity / soil_water_content) / log(soil_field_capacity / soil_wilting_point) * log(1.5 / 0.033)) * 1e3; // This last term converts from MPa to kPa.
88  double hydraulic_conductivity = soil_saturated_conductivity * pow(soil_air_entry / soil_matric_potential, 2 + 3 / soil_b_coefficient); // kg s / m^3.
89 
90  double drainage = - hydraulic_conductivity * g / density_of_water_at_20_celcius; // m / s.
91 
92  constexpr double runoff_rate = 1 / 3600; // Runoff 1 m^3 / hr.
93  double runoff = std::min(0.0, soil_water_content - soil_saturation_capacity) * runoff_rate * soil_depth; // m / s.
94 
95  double transpiration_rate = canopy_transpiration_rate / density_of_water_at_20_celcius * 1000 / 10000 / 3600; // m / s.
96  double evaporation_rate = soil_evaporation_rate / density_of_water_at_20_celcius; // m / s.
97 
98  // Update the output parameter list
99  update(soil_water_content_op, (precipitation_rate - transpiration_rate - evaporation_rate - runoff - drainage) / soil_depth * 3600); // m^3 / m^3 / hr;
100 }

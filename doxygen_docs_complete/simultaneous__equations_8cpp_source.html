172  // Make sorted lists of subsets of quantity names

174  // sorted, because a string_set is always sorted by definition.
175  string_vector steady_state_output_names = string_set_to_string_vector(find_unique_module_outputs(std::vector<string_vector>{ss_module_names}));
176  string_vector known_quantity_names = keys(known_quantities);
177  std::sort(known_quantity_names.begin(), known_quantity_names.end());
178

180  // These pointers allow the user to update the values of the known

182  // Here it is important to use sorted vectors of names so the
183  // quantities will have a definite order that the user can safely
184  // assume.
185  known_ptrs = get_pointers(known_quantity_names, quantities);
186  output_ptrs = get_const_pointers(steady_state_output_names, quantities);
187 
188  // Get vectors of pointers to important subsets of the quantities.
189  // These pointers allow us to efficiently alter portions of the
190  // quantity maps before running the modules.
191  unknown_ptrs = get_pointers(unknown_quantities, quantities);
192  steady_state_ptrs = get_pointers(steady_state_output_names, module_output_map);

184 void multilayer_canopy_properties::run() const
185 {
186  // Calculate PAR levels throughout the canopy. Note that sunML specifies that its
187  // `Idir` and `Idiff` arguments should have [micromol / m^2 / s] units. However,
188  // we can leave these quantities as energy flux densities with [J / m^2 / s] units
189  // because the two units are simply related by a multiplicative conversion factor
190  // and the output of sunML is linear with respect to `Idir` and `Idiff`.
191  struct Light_profile par_profile = sunML(par_incident_direct, par_incident_diffuse,
192  lai, nlayers, cosine_zenith_angle, kd, chil, heightf);
193 
194  // Calculate relative humidity levels throughout the canopy
195  double relative_humidity_profile[nlayers];
196  RHprof(rh, nlayers, relative_humidity_profile); // Modifies relative_humidity_profile.
197 
198  // Calculate windspeed throughout the canopy
199  double wind_speed_profile[nlayers];
200  WINDprof(windspeed, lai, nlayers, wind_speed_profile); // Modifies wind_speed_profile.
201 
202  // Calculate leaf nitrogen throughout the canopy
203  double leafN_profile[nlayers];
204  LNprof(LeafN, lai, nlayers, kpLN, leafN_profile); // Modifies leafN_profile.
205 
206  // Don't calculate anything based on the nitrogen profile
207  if (lnfun != 0) {
208  throw std::logic_error("Thrown by the multilayer_canopy_properties module: lnfun != 0 is not yet supported.");
209  }
210 
211  // Update layer-dependent outputs
212  for (int i = 0; i < nlayers; ++i) {
213  update(sunlit_fraction_ops[i], par_profile.sunlit_fraction[i]);
214  update(shaded_fraction_ops[i], par_profile.shaded_fraction[i]);
215  update(height_ops[i], par_profile.height[i]);
216  update(sunlit_incident_par_ops[i], par_profile.direct_irradiance[i]);
217  update(incident_scattered_par_ops[i], par_profile.scattered_irradiance[i]);
218  update(shaded_incident_par_ops[i], par_profile.diffuse_irradiance[i]);
219  update(incident_average_par_ops[i], par_profile.total_irradiance[i]);
220  update(rh_ops[i], relative_humidity_profile[i]);
221  update(windspeed_ops[i], wind_speed_profile[i]);
222  update(LeafN_ops[i], leafN_profile[i]);
223  }
224 }

57 void penman_monteith_transpiration::do_operation() const {
58  // Collect input parameters and make calculations
59 
60  // From Thornley and Johnson 1990. pg 408. equation 14.4k.
61  double slope_water_vapor = *slope_water_vapor_ip; // kg / m^3 / K
62  double psychr_parameter = *psychrometric_parameter_ip; // kg / m^3 / K
63  double LHV = *latent_heat_vaporization_of_water_ip; // J / kg
64  double ga = *leaf_boundary_layer_conductance_ip; // m / s
65 
66  double volume_of_one_mole_of_air = 24.39e-3; // m^3 / mol. TODO: This is for about 20 degrees C at 100000 Pa. Change it to use the model state. (1 * R * temperature) / pressure
67  double gc = *leaf_boundary_layer_conductance_ip * 1e-3 * volume_of_one_mole_of_air; // m / s
68 
69  double PhiN = *leaf_net_irradiance_ip; // In priestley_transpiration, PhiN is retrieved from "PhiN" (EBL)
70 
71  double vapor_density_deficit = *vapor_density_deficit_ip;
72  double evapotranspiration = (slope_water_vapor * PhiN + LHV * psychr_parameter * ga * vapor_density_deficit) / (LHV * (slope_water_vapor + psychr_parameter * (1 + ga / gc)));
73 
74  // Update the output parameter list
75  update(leaf_transpiration_rate_op, evapotranspiration); // kg / m^2 / s. Leaf area basis.
76 }

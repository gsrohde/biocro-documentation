126 void ed_canac_leaf::do_operation() const
127 {
128  // Get the inputs and convert units as necessary
129  const double Qp = *collatz_PAR_flux_ip * 1e6; // micromole / m^2 / s
130  const double temperature_air = *temperature_air_ip; // deg. C
131  const double RH = *rh_ip; // dimensionless
132  const double vmax = *collatz_vmax_ip * 1e6; // micromole / m^2 / s
133  const double alpha = *collatz_alpha_ip; // dimensionless
134  const double kparm = *collatz_k_ip; // mol / m^2 / s
135  const double theta = *collatz_theta_ip; // dimensionless
136  const double beta = *collatz_beta_ip; // dimensionless
137  const double Rd = *collatz_rd_ip * 1e6; // micromole / m^2 / s
138  const double bb0 = *ball_berry_intercept_ip; // mol / m^2 / s
139  const double bb1 = *ball_berry_slope_ip; // dimensionless
140  const double StomaWS = *StomataWS_ip; // dimensionless
141  const double Ca = *mole_fraction_co2_atmosphere_ip * 1e6; // micromole / mol
142  const double upperT = *collatz_rubisco_temperature_upper_ip; // deg. C
143  const double lowerT = *collatz_rubisco_temperature_lower_ip; // deg. C
144  const int water_stress_approach = 1; // Apply water stress via stomatal conductance
145  const double Rad = 0.0; // micromoles / m^2 / s (only used for evapotranspiration)
146  const double WindSpeed = *windspeed_ip; // m / s
147  const double LeafAreaIndex = 0.0; // dimensionless from m^2 / m^2 (not actually used by EvapoTrans2)
148  const double CanopyHeight = 0.0; // meters (not actually used by EvapoTrans2)
149  const double leaf_width = *leafwidth_ip; // meter
150  const int eteq = 0; // Report Penman-Monteith transpiration
151 
152  // For ed_penman_monteith_leaf_temperature, the light input is called `solar_energy_absorbed_leaf`
153  // and is in units of W / m^2 / s. This quantity is equivalent to `Ja2` in Evapotrans2. `Iave` is
154  // related to `Ja2` by:
155  // Ja2 = total_average_irradiance * (1 - LeafReflectance - tau) / (1 - tau)
156  // = Iave * joules_per_micromole_PAR / fraction_of_irradiance_in_PAR * (1 - LeafReflectance - tau) / (1 - tau)
157  // where:
158  // joules_per_micromole_PAR = 0.235
159  // fraction_of_irradiance_in_PAR = 0.5
160  // LeafReflectance = 0.2
161  // tau = 0.2
162  // So overall, Ja2 = Iave * 0.235 / 0.5 * (1 - 0.2 - 0.2) / (1 - 0.2)
163  // = Iave * 0.3525
164  // So we can find Iave according to Iave = Ja2 / 0.3525
165  const double Iave = *solar_energy_absorbed_leaf_ip / 0.3525; // micromoles / m^2 / s
166 
167  // Run c4photoC once assuming the leaf is at air temperature to get an initial guess for gs
168  const struct c4_str first_c4photoC_output = c4photoC(Qp, temperature_air, RH,
169  vmax, alpha, kparm, theta, beta, Rd,
170  bb0, bb1, StomaWS, Ca, water_stress_approach,
171  upperT, lowerT);
172 
173  const double stomatal_conductance = first_c4photoC_output.Gs; // mmol / m^2 / s
174 
175  // Run EvapoTrans2 to get the leaf temperature
176  const struct ET_Str evapotrans_output = EvapoTrans2(Rad, Iave, temperature_air, RH, WindSpeed,
177  LeafAreaIndex, CanopyHeight, stomatal_conductance, leaf_width, eteq);
178 
179  const double leaf_temperature = temperature_air + evapotrans_output.Deltat;
180 
181  // Run c4photoC a second time to get final values for assimilation and conductance
182  const struct c4_str second_c4photoC_output = c4photoC(Qp, leaf_temperature, RH,
183  vmax, alpha, kparm, theta, beta, Rd,
184  bb0, bb1, StomaWS, Ca, water_stress_approach,
185  upperT, lowerT);
186 
187  // Convert and return the results
188  update(mole_fraction_co2_intercellular_op, second_c4photoC_output.Ci * 1e-6); // mol / m^2 / s
189  update(conductance_stomatal_h2o_op, second_c4photoC_output.Gs * 1e-3); // mol / m^2 / s
190  update(assimilation_gross_op, second_c4photoC_output.GrossAssim * 1e-6); // mol / m^2 / s
191  update(assimilation_net_op, second_c4photoC_output.Assim * 1e-6); // mol / m^2 / s
192  update(evapotranspiration_penman_monteith_op, evapotrans_output.TransR * 1e-3); // mol / m^2 / s
193  update(evapotranspiration_penman_op, evapotrans_output.EPenman * 1e-3); // mol / m^2 / s
194  update(evapotranspiration_priestly_op, evapotrans_output.EPriestly * 1e-3); // mol / m^2 / s
195  update(temperature_leaf_op, leaf_temperature); // deg. C
196 }

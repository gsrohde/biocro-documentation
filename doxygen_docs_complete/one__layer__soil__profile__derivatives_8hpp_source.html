75 void one_layer_soil_profile_derivatives::do_operation() const {
76  // Collect inputs and make calculations
77  double soil_water_content = *soil_water_content_ip; // m^3 m^-3.
78  double soil_depth = *soil_depth_ip;
79  double soil_field_capacity = *soil_field_capacity_ip;
80  double soil_wilting_point = *soil_wilting_point_ip;
81  double soil_saturated_conductivity = *soil_saturated_conductivity_ip;
82  double soil_air_entry = *soil_air_entry_ip;
83  double soil_b_coefficient = *soil_b_coefficient_ip;
84  double acceleration_from_gravity = *acceleration_from_gravity_ip;
85  double precipitation_rate = *precipitation_rate_ip;
86  double soil_saturation_capacity = *soil_saturation_capacity_ip;
87  double soil_sand_content = *soil_sand_content_ip;
88  double evapotranspiration = *evapotranspiration_ip;
89 
90  double soil_water_potential = -exp(log(0.033) + ((log(soil_field_capacity) - log(soil_water_content)) / (log(soil_field_capacity) - log(soil_wilting_point)) * (log(1.5) - log(0.033)))) * 1e3; // kPa.
91 
92  double hydraulic_conductivity = soil_saturated_conductivity * pow(soil_air_entry / soil_water_potential, 2 + 3 / soil_b_coefficient); // Kg s m^-3.
93  double J_w = -hydraulic_conductivity * (-soil_water_potential / (soil_depth * 0.5)) - acceleration_from_gravity * hydraulic_conductivity; // kg m^-2 s^-1. Campbell, pg 129. I multiply soil depth by 0.5 to calculate the average depth.
94 
95  double density_of_water_at_20_celcius = 998.2; // kg m^-3
96 
97  double drainage = J_w / density_of_water_at_20_celcius; // m s^-1.
98  double precipitation_m_s = precipitation_rate * 1e-3; // m s^-2.
99 
100  double second_per_hour = 3600;
101  double runoff = (soil_water_content - soil_saturation_capacity) * soil_depth / second_per_hour; // m s^-1. The previous model drained everything in a single time-step. By default the time step was one hour, so use a rate that drains everything in one hour.
102  double n_leach = runoff / 18 * (0.2 + 0.7 * soil_sand_content) / second_per_hour; // Base the rate on an hour for the same reason as was used with `runoff`.
103 
104  double evapotranspiration_volume = evapotranspiration / density_of_water_at_20_celcius / 1e4 / second_per_hour; // m^3 m^-2 s^-1
105 
106  // Update the output parameter list
107  update(soil_water_content_op, (drainage + precipitation_m_s - runoff - evapotranspiration_volume) / soil_depth);
108  update(soil_n_content_op, -n_leach);
109 }

131 void grimm_soybean_flowering_calculator::do_operation() const
132 {
133  // Collect inputs
134  const double grimm_sowing_doy = *grimm_sowing_doy_ip; // calendar day of year
135  const double grimm_physiological_age = *grimm_physiological_age_ip; // physiological days
136  const double grimm_juvenile_T0 = *grimm_juvenile_T0_ip; // deg. C
137  const double grimm_juvenile_T1 = *grimm_juvenile_T1_ip; // deg. C
138  const double grimm_juvenile_T2 = *grimm_juvenile_T2_ip; // deg. C
139  const double grimm_juvenile_T3 = *grimm_juvenile_T3_ip; // deg. C
140  const double grimm_juvenile_pd_threshold = *grimm_juvenile_pd_threshold_ip; // physiological days
141  const double grimm_T_min = *grimm_T_min_ip; // deg. C
142  const double grimm_T_opt = *grimm_T_opt_ip; // deg. C
143  const double grimm_N_min = *grimm_N_min_ip; // hours
144  const double grimm_N_opt = *grimm_N_opt_ip; // hours
145  const double grimm_flowering_threshold = *grimm_flowering_threshold_ip; // physiological days
146  const double doy_dbl = *doy_dbl_ip; // fractional day of year
147  const double temp = *temp_ip; // deg. C
148  const double day_length = *day_length_ip; // hours
149 
150  // Calculate the night length, applying a correction to convert from astronomical day
151  constexpr double day_length_correction = 1.0; // dimensionless (from effective day length / celestial day length). Set to 0.883 to use the effective day length from Goudriaan (1978).
152  const double night_length = 24.0 - day_length * day_length_correction; // hours
153 
154  // Calculate the normalized growth rate based on the day of year, night length, temperature, and growth stage
155  double rate_normalized = 0.0; // dimensionless
156  double temperature_rate_normalized = 0.0; // dimensionless
157  double photoperiod_rate_normalized = 0.0; // dimensionless
158  if (doy_dbl < grimm_sowing_doy) {
159  // The plant has not been sowed yet, so it can't grow
160  rate_normalized = 0.0;
161  } else if (grimm_physiological_age < grimm_juvenile_pd_threshold) {
162  // The plant is still in the juvenile (photoperiod-insensitive) phase
163  if (temp < grimm_juvenile_T0) {
164  // The temperature is too low for the plant to grow
165  rate_normalized = 0.0;
166  } else if (temp < grimm_juvenile_T1) {
167  // There is a linear increase from 0 to 1 as temp changes from T0 to T1
168  rate_normalized = (temp - grimm_juvenile_T0) / (grimm_juvenile_T1 - grimm_juvenile_T0);
169  } else if (temp < grimm_juvenile_T2) {
170  // We are in the flat region at maximum growth
171  rate_normalized = 1.0;
172  } else if (temp < grimm_juvenile_T3) {
173  // There is a linear decrease from 1 to 0 as temp changes from T2 to T3
174  rate_normalized = 1.0 - (temp - grimm_juvenile_T2) / (grimm_juvenile_T3 - grimm_juvenile_T2);
175  } else {
176  // The temperature is too high for the plant to grow
177  rate_normalized = 0.0;
178  }
179  } else {
180  // The plant has entered the photoperiod-sensitive phase
181  if (temp < grimm_T_min) {
182  // The temperature is too low for the plant to develop
183  temperature_rate_normalized = 0.0;
184  } else if (temp < grimm_T_opt) {
185  // There is a linear increase from 0 to 1 as temp changes from T_min to T_opt
186  temperature_rate_normalized = (temp - grimm_T_min) / (grimm_T_opt - grimm_T_min);
187  } else {
188  // The development rate is maxed out at 1
189  temperature_rate_normalized = 1.0;
190  }
191  if (night_length < grimm_N_min) {
192  // The night is too short for the plant to develop
193  photoperiod_rate_normalized = 0.0;
194  } else if (night_length < grimm_N_opt) {
195  // There is a linear increase from 0 to 1 as night_length changes from N_min to N_opt
196  photoperiod_rate_normalized = (night_length - grimm_N_min) / (grimm_N_opt - grimm_N_min);
197  } else {
198  // The development rate is maxed out at 1
199  photoperiod_rate_normalized = 1.0;
200  }
201  rate_normalized = temperature_rate_normalized * photoperiod_rate_normalized;
202  }
203 
204  // Decide if the plant is flowering
205  const bool is_flowering = (grimm_physiological_age > grimm_juvenile_pd_threshold + grimm_flowering_threshold);
206 
207  // Scale the rates to the maximum value, which is 1 physiological day per day = 1 / 24 physiological days per hour
208  const double rate_per_hour = rate_normalized / 24.0;
209  const double temperature_rate_per_hour = temperature_rate_normalized / 24.0;
210  const double photoperiod_rate_per_hour = photoperiod_rate_normalized / 24.0;
211 
212  // Update the output parameter list
213  update(grimm_rate_op, rate_per_hour);
214  update(grimm_rate_temperature_op, temperature_rate_per_hour);
215  update(grimm_rate_photoperiod_op, photoperiod_rate_per_hour);
216  update(grimm_flowering_op, is_flowering);
217 }

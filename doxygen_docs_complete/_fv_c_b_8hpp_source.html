100  double leaf_stomatal_conductance = *leaf_stomatal_conductance_ip;
101 
102  double rubisco_limited = Vcmax * (Ci - Gstar) / (Ci + Kc * (1.0 + O2 / Ko));
103 
104  double rubp_limited = J * (Ci - Gstar) / (electrons_per_carboxylation * Ci + 2.0 * electrons_per_oxygenation * Gstar);
105  rubp_limited = std::max(rubp_limited, 0.0); // This rate can't be negative.
106 
107  double tpu_limited = 3.0 * maximum_tpu_rate / (1.0 - Gstar / Ci);
108 
109  double carboxylation_rate = std::min(rubisco_limited, std::min(rubp_limited, tpu_limited)); // The overall carboxylation rate is the rate of the slowest process.
110  double net_assimilation_rate = carboxylation_rate - Rd;
111 
112  Ci = Ca - net_assimilation_rate * 1.6 * atmospheric_pressure / leaf_stomatal_conductance;
113 
114  // Update the output parameter list
115  update(carboxylation_rate_op, carboxylation_rate);
116  update(net_assimilation_rate_op, net_assimilation_rate);

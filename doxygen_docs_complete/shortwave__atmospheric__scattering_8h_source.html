75 void shortwave_atmospheric_scattering::do_operation() const
76 {
77  // Calculate diffuse and direct irradiance transmittance
78  double irradiance_direct_transmittance;
79  double irradiance_diffuse_transmittance;
80  if (cosine_zenith_angle <= 0) {
81  // In this case, the sun is at or below the horizon, so there is no direct irradiance
82  irradiance_direct_transmittance = 0.0;
83  irradiance_diffuse_transmittance = 1.0;
84  } else {
85  // The sun is above the horizon, so use its zenith angle to calculate transmittance
86  // The direct transmission is calculated using Equation 11.11 from Campbell & Norman (1998), solving for
87  // irradiance_direct_transmittance = S_p / S_p0
88  // The indirect (AKA diffuse) transmission is calculated using Equation 11.13 from Campbell & Norman (1998), solving for
89  // irradiance_diffuse_transmittance = S_d / S_p0
90  const double pressure_ratio = atmospheric_pressure / physical_constants::pressure_at_sea_level; // dimensionless.
91  irradiance_direct_transmittance = pow(atmospheric_transmittance, (pressure_ratio / cosine_zenith_angle)); // Modified from equation 11.11 of Norman and Campbell.
92  irradiance_diffuse_transmittance = atmospheric_scattering * (1.0 - irradiance_direct_transmittance) * cosine_zenith_angle; // Modified from equation 11.13 of Norman and Campbell.
93  }
94 
95  // Calculate the diffuse and direct fractions
96  const double irradiance_direct_fraction = irradiance_direct_transmittance / (irradiance_direct_transmittance + irradiance_diffuse_transmittance);
97  const double irradiance_diffuse_fraction = 1.0 - irradiance_direct_fraction;
98 
99  // Update the outputs
100  update(irradiance_direct_transmittance_op, irradiance_direct_transmittance);
101  update(irradiance_diffuse_transmittance_op, irradiance_diffuse_transmittance);
102  update(irradiance_direct_fraction_op, irradiance_direct_fraction);
103  update(irradiance_diffuse_fraction_op, irradiance_diffuse_fraction);
104 }

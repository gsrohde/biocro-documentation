99 void flowering::do_operation() const {
100  // Collect inputs and make calculations
101  double solar = *solar_ip;
102  double doy_dbl = *doy_dbl_ip;
103  double dawn = *dawn_ip; // In hours; a value of 100 means it's currently night. Other values indicate dawn for that day.
104  double GI_on = *GI_on_ip;
105  double GI_off = *GI_off_ip;
106  double FKF1_on = *FKF1_on_ip;
107  double FKF1_off = *FKF1_off_ip;
108  double CDF_on = *CDF_on_ip;
109  double CDF_off = *CDF_off_ip;
110  double FT_on = *FT_on_ip;
111  double FT_off = *FT_off_ip;
112  double FT_inhibition = *FT_inhibition_ip;
113  double FKF1_timing = *FKF1_timing_ip;
114  double CDF_timing = *CDF_timing_ip;
115  double GI = *GI_ip;
116  double FKF1 = *FKF1_ip;
117  double CDF = *CDF_ip;
118  double FT = *FT_ip;
119 
120  // Get the hour from the fractional part of doy_dbl
121  double hour = 24.0 * (doy_dbl - floor(doy_dbl));
122 
123  // Initialize the derivatives
124  double d_GI, d_FKF1, d_CDF, d_FT;
125 
126  // FKF1 starts being produced at a rate proportional to the solar radiation strength
127  // a certain number of hours after dawn (set by FKF1_timing)
128  // FKF1 also has an exponential decay term that is always present, regardless of
129  // sunlight or dawn
130  if(hour > dawn + FKF1_timing) d_FKF1 = FKF1_on * solar - FKF1_off * FKF1;
131  else d_FKF1 = - FKF1_off * FKF1;
132 
133  // GI starts being produced at a rate proportional to the solar radiation strength
134  // three hours after FKF1 starts being produced (hard-coded)
135  // GI also has an exponential decay term that is always present, regardless of
136  // sunlight or dawn
137  if(hour > dawn + FKF1_timing + 3.0) d_GI = GI_on * solar - GI_off * GI;
138  else d_GI = - GI_off * GI;
139 
140  // CDF starts being produced at a rate proportional to the solar radiation strength
141  // a certain number of hours after dawn (set by CDF_timing)
142  // CDF has an exponential decay term that is always present and is accelerated by
143  // the simultaneous presence of FKF1 and GI
144  if(hour > dawn + CDF_timing) d_CDF = CDF_on * solar - CDF_off * (1 + GI * FKF1) * CDF;
145  else d_CDF = - CDF_off * (1 + GI * FKF1) * CDF;
146 
147  // FT is produced at a constant rate in the absence of CDF, but the CDF inhibits FT production
148  // by decreasing this rate
149  // FT has an exponential decay term that is always present
150  d_FT = FT_on / (1 + FT_inhibition * CDF) - FT_off * FT;
151 
152  // Update the output parameter list
153  update(GI_op, d_GI);
154  update(FKF1_op, d_FKF1);
155  update(CDF_op, d_CDF);
156  update(FT_op, d_FT);
157 }

21 quantity<dimensionless> arrhenius_exponent(quantity<dimensionless> c, quantity<energy_over_amount> activation_energy, quantity<temperature> temperature)
22 {
23  const quantity<energy_over_temperature_amount> R = 8.314472 * joule / kelvin / mole ;
24  return exp(c - activation_energy / (R * temperature));
25 }

30 {
31  // Assign units to the input parameters. The parameters can be renamed and this section can be removed when call functions
32  // that call c3photoC() are also using units.
33  const quantity<flux> Rd0 = _Rd0 * 1e-6 * mole / square_meter / second;
34  const quantity<flux> Vcmax0 = _Vcmax0 * 1e-6 * mole / square_meter / second;
35  const quantity<mole_fraction> atmospheric_oxygen_mole_fraction = _O2 * 1e-3 * mole / mole;
36  const quantity<flux> Jmax = _Jmax * 1e-6 * mole / square_meter / second;
37  const quantity<temperature> leaf_temperature = (_Tleaf + 273.15) * kelvin;
38  const quantity<flux> Qp = _Qp * 1e-6 * mole / square_meter / second;
39 
40  const quantity<pressure> atmospheric_pressure = 101325 * pascal;
41  const quantity<dimensionless> leaf_reflectance = 0.2;
42  const quantity<flux> maximum_tpu_rate = 23 * 1e-6 * mole / square_meter / second;
43 
44  /* From Bernacchi 2001. Improved temperature response functions. */
45  /* Note: Values in Dubois and Bernacchi are incorrect. */
46  const quantity<mole_fraction> Kc = 1e-6 * arrhenius_exponent(38.05, 79.43e3 * joule / mole, leaf_temperature);
47  const quantity<mole_fraction> Ko = 1e-3 * arrhenius_exponent(20.30, 36.38e3 * joule / mole, leaf_temperature);
48  const quantity<mole_fraction> Gstar = 1e-6 * arrhenius_exponent(19.02, 37.83e3 * joule / mole, leaf_temperature);
49  const quantity<flux> Vcmax = Vcmax0 * arrhenius_exponent(26.35, 65.33e3 * joule / mole, leaf_temperature);
50  const quantity<flux> Rd = Rd0 * arrhenius_exponent(18.72, 46.39e3 * joule / mole, leaf_temperature);
51 
52  const double leaf_temperature_celsius = leaf_temperature.value() - 273.15;
53  const quantity<dimensionless> theta = thet + 0.018 * leaf_temperature_celsius - 3.7e-4 * pow(leaf_temperature_celsius, 2);
54 
55  /* Light limited */
56  const quantity<dimensionless> dark_adapted_phi_PSII = 0.352 + 0.022 * leaf_temperature_celsius - 3.4 * pow(leaf_temperature_celsius, 2) / 10000; // Bernacchi et al. 2003 Plant, Cell and Environment 26, 14191430 doi: 10.1046/j.0016- 8025.2003.01050.x
57  const quantity<flux> I2 = Qp * dark_adapted_phi_PSII * (1.0 - leaf_reflectance) / 2.0;
58 
59  const quantity<flux> J = (Jmax + I2 - root<2>(pow<2>(Jmax + I2) - 4.0 * theta * I2 * Jmax )) / (2.0 * theta);
60 
61  const quantity<mole_fraction> Oi = atmospheric_oxygen_mole_fraction * solo(leaf_temperature_celsius);
62 
63  if(Ca <= 0)
64  Ca = 1e-4;
65 
66  const quantity<pressure> Ca_pa = Ca * 1e-6 * atmospheric_pressure; // Pa.
67 
68  quantity<pressure> Ci_pa = 0.0 * pascal;
69  quantity<flux> Vc;
70  quantity<flux> Tol = 0.01 * 1e-6 * mole / square_meter / second;
71  quantity<flux> Gs;
72  quantity<flux> co2_assimilation_rate = 0 * mole / square_meter / second;
73  quantity<mole_fraction> Ci;
74  int iterCounter = 0;
75  int max_iter = 1000;
76  while (iterCounter < max_iter) {
77  quantity<flux> OldAssim = co2_assimilation_rate;
78 
79  /* Rubisco limited carboxylation */
80  Ci = Ci_pa / atmospheric_pressure;
81  quantity<flux> Ac1 = Vcmax * (Ci - Gstar);
82  double Ac2 = Ci + Kc * (1 + Oi / Ko);
83  quantity<flux> Ac = Ac1 / Ac2;
84 
85  /* Light lmited portion */
86  quantity<flux> Aj1 = J * (Ci - Gstar);
87  double Aj2 = electrons_per_carboxylation * Ci + 2.0 * electrons_per_oxygenation * Gstar;
88  quantity<flux> Aj = Aj1 / Aj2;
89  if (Aj < 0.0 * mole / square_meter / second)
90  Aj = 0.0 * mole / square_meter / second;
91 
92  /* Limited by tri phos utilization */
93  quantity<flux> Ap = 3.0 * maximum_tpu_rate / (1 - Gstar / Ci);
94 
95  if(Ac < Aj && Ac < Ap){
96  Vc = Ac;
97  } else if (Aj < Ac && Aj < Ap){
98  Vc = Aj;
99  } else if (Ap < Ac && Ap < Aj){
100  if (Ap < 0 * mole / square_meter / second)
101  Ap = 0 * mole / square_meter / second;
102  Vc = Ap;
103  }
104 
105  co2_assimilation_rate = Vc - Rd;
106 
107  if (water_stress_approach == 0) co2_assimilation_rate *= quantity<dimensionless>(StomWS);
108  /* milimole per meter square per second*/
109  Gs = ball_berry(co2_assimilation_rate.value(), Ca * 1e-6, RH, bb0, bb1) * 1e-3 * mole / square_meter / second;
110 
111  if (water_stress_approach == 1)
112  //Gs *= quantity<dimensionless>(StomWS);
113  Gs *= StomWS;
114 
115  if (Gs <= 0 * mole / square_meter / second)
116  Gs = 1e-5 * 1e-3 * mole / square_meter / second;
117 
118  if (Gs > 800 * 1e-3 * mole / square_meter / second)
119  Gs = 800 * 1e-3 * mole / square_meter / second;
120 
121  Ci_pa = Ca_pa - co2_assimilation_rate * 1.6 * atmospheric_pressure / Gs;
122 
123  if (Ci_pa < 0 * pascal)
124  Ci_pa = 1e-5 * pascal;
125 
126  if (abs(OldAssim - co2_assimilation_rate) < Tol) {
127  break;
128  }
129 
130  ++iterCounter;
131  }
132  if(iterCounter == max_iter) {
133  /*
134  // The rate failed to converge! Rerun in debug mode, then throw an error
135  c3_str result;
136  result = c3photoCdb(_Qp, _Tleaf, RH, _Vcmax0, _Jmax,
137  _Rd0, bb0, bb1, Ca, _O2,
138  thet, StomWS, water_stress_approach, electrons_per_carboxylation, electrons_per_oxygenation);
139  throw std::range_error("Thrown in c3photo: co2_assimilation_rate did not converge");
140  */
141 
142  //std::cout << "co2_assimilation_rate did not converge\n";
143  }
144  struct c3_str result;
145  result.Assim = co2_assimilation_rate.value() * 1e6; // micromole / m^2 / s.
146  result.Gs = Gs.value() * 1e3; // mmol / m^2 / s.
147  result.Ci = Ci.value() * 1e6; // micromole / m^2 / s.
148  result.GrossAssim = (co2_assimilation_rate.value() + Rd.value()) * 1e6; // micromole / m^2 / s.
149  return result;
150 }

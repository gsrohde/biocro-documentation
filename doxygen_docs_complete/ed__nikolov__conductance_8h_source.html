127 void ed_nikolov_conductance_forced::do_operation() const
128 {
129  // Convert temperatures to Kelvin
130  const double Tak = *temperature_air_ip + physical_constants::celsius_to_kelvin; // Kelvin
131 
132  // Calculate the volume of one mole of a gas at air temperature and pressure
133  // using the ideal gas law
134  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / *atmospheric_pressure_ip; // m^3 / mol
135 
136  // Calculate the forced boundary layer conductance for water vapor
137  // using Equation 29 in Nikolov et al. Note that we have used
138  // volume_per_mol to convert from m / s to mol / m^2 / s (see
139  // class description above for more details).
140  const double gbv_forced = *nikolov_cf_ip *
141  pow(Tak, nikolov::temperature_exponent) *
142  pow((Tak + nikolov::temperature_offset) * *windspeed_ip / (*leafwidth_ip * *atmospheric_pressure_ip), 0.5) /
143  volume_per_mol; // mol / m^2 / s
144 
145  // Check for error conditions
146  std::map<std::string, bool> errors_to_check = {
147  {"atmospheric_pressure cannot be zero", *atmospheric_pressure_ip == 0}, // divide by zero
148  {"volume_per_mol cannot be zero", volume_per_mol == 0}, // divide by zero
149  {"leaf_width cannot be zero", *leafwidth_ip == 0} // divide by zero
150  };
151 
152  check_error_conditions(errors_to_check, get_name());
153 
154  // Update the output parameter list
155  update(conductance_boundary_h2o_forced_op, gbv_forced);
156 }

231 void ed_nikolov_conductance_free::do_operation() const
232 {
233  // Convert temperatures to Kelvin
234  const double Tak = *temperature_air_ip + physical_constants::celsius_to_kelvin; // Kelvin
235  const double Tlk = *temperature_leaf_ip + physical_constants::celsius_to_kelvin; // Kelvin
236 
237  // Calculate the volume of one mole of a gas at air temperature and pressure
238  // using the ideal gas law
239  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / *atmospheric_pressure_ip; // m^3 / mol
240 
241  // Calculate the water vapor mole fraction at the leaf surface that would occur
242  // if boundary_layer_conductance = free_boundary_layer_conductance. This calculation
243  // is analogous to Equation 35 in Nikolov et al., which was originally derived by
244  // requiring equal energy fluxes in the boundary layer and across the stomata. Here
245  // we use molecular fluxes instead. Also note that our concentration at the leaf
246  // surface is equivalent to Nikolov's concentration in the boundary layer.
247  const double mole_fraction_h2o_intercellular = saturation_vapor_pressure(*temperature_leaf_ip) / *atmospheric_pressure_ip; // dimensionless
248  const double h2o_mfs_numer = *conductance_boundary_h2o_free_ip * *mole_fraction_h2o_atmosphere_ip +
249  *conductance_stomatal_h2o_ip * mole_fraction_h2o_intercellular;
250  const double h2o_mfs_denom = *conductance_boundary_h2o_free_ip + *conductance_stomatal_h2o_ip;
251  const double mole_fraction_h2o_leaf_surface = h2o_mfs_numer / h2o_mfs_denom; // dimensionless
252 
253  // Calculate the virtual temperature difference using Equation 34 of Nikolov et al.
254  // Note that the original equation contains terms of partial_pressure/atmospheric_pressure;
255  // We have replaced those terms with equivalent mole fractions.
256  const double Tvdiff = Tlk / (1.0 - nikolov::Tvdiff_factor * mole_fraction_h2o_leaf_surface) -
257  Tak / (1.0 - nikolov::Tvdiff_factor * *mole_fraction_h2o_atmosphere_ip); // deg. C or K
258 
259  // Calculate the free boundary layer conductance using Equation 33 from Nikolov et al.
260  // Note that we have used volume_per_mol to convert from m / s to mol / m^2 / s (see
261  // class description above for more details).
262  const double gbv_free = *nikolov_ce_ip *
263  pow(Tlk, nikolov::temperature_exponent) *
264  pow((Tlk + nikolov::temperature_offset) / *atmospheric_pressure_ip, 0.5) *
265  pow(fabs(Tvdiff) / *leafwidth_ip, 0.25) /
266  volume_per_mol; // mol / m^2 / s
267 
268  // Check for error conditions
269  std::map<std::string, bool> errors_to_check = {
270  {"atmospheric_pressure cannot be zero", *atmospheric_pressure_ip == 0}, // divide by zero
271  {"volume_per_mol cannot be zero", volume_per_mol == 0}, // divide by zero
272  {"leaf_width cannot be zero", *leafwidth_ip == 0}, // divide by zero
273  {"the sum of water vapor conductances cannot be zero", h2o_mfs_denom == 0} // divide by zero
274  };
275 
276  check_error_conditions(errors_to_check, get_name());
277 
278  // Update the output parameter list
279  update(conductance_boundary_h2o_free_op, gbv_free);
280 }
